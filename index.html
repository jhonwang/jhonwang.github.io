帐号某功能

function init(){
    var mod = new Mod({
        validateForType:{
            'vcode':/^[0-9]{6}$/,
            'phone':/^1[34578][0-9]{9}$/,
            'pwd':function(target,val,errorDom){
                if(val == ''){
                    this.showError(errorDom,'密码不能为空');
                    return !1;
                }else if(val.length < 6 || val.length >16){
                    this.showError(errorDom,'密码为6-16位');
                    return !1;
                }
                return !0;
            }
        },
        modBtnList:{
            'phone':'J-phone',
            'email':'J-email'
        },
        eventsForId:{
            'J-check-vcode':function(target){
                this.submitInterface(target,'url',null,!1,!1,'phone');
            },
            'J-phone':function(target){
                this.submitInterface(target,'url')
            }
        },
        responseFunForId:{
            'J-check-vcode':function(data,target,errorDom){
                if(data.retcode == '20000000'){
                    Tools.addClass(target,'isDisable');
                    mod.resendLoop(target,120)
                }else{
                    mod.showError(errorDom,data.msg);
                }
            },
            'J-phone':normalSuccessFun
        }
    });

    // 成功处理函数
    function normalSuccessFun(data,target,errorDom){
        if(data && data.retcode == 2e7 && data.data){
            location.href = data.data.url;
        }else{
            this.showError(errorDom,data.msg?data.msg:'系统繁忙，稍后重试');
        }
    }

}
window.addEventListener("DOMContentLoaded", init, false);

@mixin border-half-px($type,$color){
  @each $var in $type {
    border-#{$var}:$color solid 1px;
  }
  @at-root{
    @media (-webkit-min-device-pixel-ratio: 2), (min-device-pixel-ratio: 2) {
      &{
        border: none;
        background-repeat: no-repeat;

        background-image:nth(half-px-processor($type,$color),2);
        background-image:nth(half-px-processor($type,$color),3);
        -wekbit-background-image:nth(half-px-processor($type,$color),4);
        background-size:nth(half-px-processor($type,$color),4);
        background-position:nth(half-px-processor($type,$color),5);
